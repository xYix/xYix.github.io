回顾匈牙利算法的流程。

我们需要执行下面的步骤 N 次，每次把所有节点变为未染色状态并从一个新的没匹配的节点开始搜索：

对于一个没匹配的黑点，枚举所有和它有边的节点：

  • 如果是一个白点，那么说明找到了一个偶环。如果把偶环上的所有边的是否匹配的属性反转，那么匹配数不会增多也不会减少，所以直接跳过。

  • 如果是一个没被染色的节点，把它染白，然后：

    • 如果它没有匹配，那么找到了一条增广路，把它上面的边全部反转可以使匹配数+1。反转，结束。

    • 如果有匹配，那么从与之匹配的黑点继续搜索。

可以发现匈牙利算法之所以在一般图上不能使用，是因为它没有讨论如果访问到了一条黑黑边该怎么办。这时会出现一个奇环，一个 2n+1 条边的奇环，内部可以做出 n 组匹配，还有一个“插头”可以向外连一条边，这个插头是可以是奇环内的任意一个点，如图：

（当然也可以牺牲内部的匹配获得更多的插头，但是不比只有一个插头好）

这样一个奇环完全就可以看做是一个点了。

那么带花树算法就是匈牙利的流程补上对于奇环的处理：

  • 如果是一个黑点，意味着出现了奇环，把这个奇环缩成一个点，把上面的白点全部染黑并塞进队列。（BFS 会使得带花树的实现容易很多）

为了更方便地找到奇环还需要加 pre[x] 表示本次增广中哪个点走一条非匹配边会走到 x（如果在奇环内，会有 pre[x]=y，pre[y]=x），这样的话暴力跳父亲就可以找到奇环了。

#include<bits/stdc++.h>
using namespace std;


const int maxN=2005,maxM=50005;
int N,M;
int match[maxN];


vector<int> G[maxN];
void add_E(int u,int v){
    G[u].push_back(v);G[v].push_back(u);
}


int pre[maxN],clr[maxN],tag[maxN],fa[maxN];
int find(int x){return (fa[x]==x?x:fa[x]=find(fa[x]));}


int q[maxN],h,t;
int cnttag;
void Reverse(int u){
    if(u) Reverse(match[pre[u]]),match[u]=pre[u],match[pre[u]]=u;
}
int Lca(int u,int v){
    cnttag++;
    for(;;u=pre[match[u]],swap(u,v))
        if(tag[u=find(u)]==cnttag) return u;
        else if(u) tag[u]=cnttag;
}
void Shrink(int u,int v,int p){
    for(;find(u)!=p;u=pre[v]){
        pre[u]=v,v=match[u],fa[u]=fa[v]=p;
        if(clr[v]==1) clr[v]=2,q[++t]=v;
    }
}
bool Solve(int stt){ 
    cnttag=0;
    for(int i=1;i<=N;i++) pre[i]=0,clr[i]=0,tag[i]=0,fa[i]=i;
    h=t=0;q[++t]=stt;clr[stt]=2;
    while(h<t){
        int u=q[++h];
        for(int i=0;i<G[u].size();i++){
            int v=G[u][i];
            if(!clr[v]){
                clr[v]=1;pre[v]=u;
                if(!match[v]) {Reverse(v);return 1;}
                else clr[match[v]]=2,q[++t]=match[v];
            }
            else if(clr[v]==2){
                int p=Lca(u,v);Shrink(u,v,p);Shrink(v,u,p);
            }
        }
    }
    return 0;
}


int main(){
    scanf("%d%d",&N,&M);
    for(int i=1;i<=M;i++){
        int u,v;scanf("%d%d",&u,&v);
        add_E(u,v);
    }
    int ans=0;
    for(int i=1;i<=N;i++) if(!match[i]) ans+=Solve(i);
    printf("%d\n",ans);
    for(int i=1;i<=N;i++) printf("%d ",match[i]);
}
