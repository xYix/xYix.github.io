记得在树链剖分中，我们把每个节点的其中一个儿子看得特殊一些处理，在树剖中，我们称之为“重儿子”，从节点到其重儿子的边称为“重边”，一串重边称为“重链”。

类似的，LCT中也会把树分为“实边”（或称“偏好边”Preferred Edge）和“虚边”，每个节点都只有一个实儿子（或称“偏好儿子”Preferred Child）。

在LCT中对于每一条实链（或称“偏好链”Preferred Chain）我们都用一棵平衡树来维护，由于这棵平衡树需要一些interestring的性质所以我们一般使用Splay。

  • 每个节点在且仅在一个Splay中。

  • 每个Splay中序遍历得到的序列就是所代表的实链从上到下的顺序。

  • 虚边链接了一条实链（称为实链a）的顶端和一条实链的中间一个节点（这个节点称为x）。我们设定a的Splay的根节点的（Splay意义上的）父亲为x，但是x并没有a的Splay的根节点这一（Splay意义上的）儿子。很多地方称这个性质为“认父不认子”。如图所示。

1.Access 操作

Access(x)操作所做的，就是把x到根的一整条路径变为实链。

过程很暴力，就是不断往上跳：

比如现在所在实链顶端是y，他的（原树上的）父亲是x（也是当前实链的根节点的Splay意义上的父亲，实际上我们还是这个Splay意义来找的）。

我们发现x−>y是虚边。于是splay(x)（方便下面的操作），然后直接把x所连向它的实儿子的实边变虚（在Splay上不要这个儿子）（请注意这个实儿子仍然认x这个父亲），并把x−>y变成实边（合并x所在实链的Splay和当前所在实链的Splay），然后继续。

代码也很短：

void Access(int x){
    for(int y=0;x;y=x,x=T[x].fa)
        splay(x),T[x].c[1]=y,push_up(x);
}

你可能会问x原来的那个实儿子怎么办，其实你仔细想想就会发现我们这么一通瞎操作之后他的信息倒是正确的。

2.MakeRoot 操作

MakeRoot(x)操作就是将x钦定为原树的根。

首先肯定先要Access(x)一下，这样会方便。

注意到Access(x)之后，x肯定是所在实链的最底端，而且并没有向儿子连出任何实边（不理解可以回去看Access的代码）。那么只需要把x所在实链的深度顺序翻转一下即可。这里reverse即为翻转函数。所以显然还需要一个push_down函数下传翻转标记。

xxxxxxxxxx
void MakeRoot(int x){
    Access(x);splay(x);reverse(x);
}

3.FindRoot 操作

FindRoot(x)操作就是找到x在原树中的根。

首先肯定先要Access(x)一下，这样会方便。然后把xsplay到根，这样根到x的路径，也是一条实链，都在（Splay中）x的左边。所以不断往左跳（跳的时候不要忘记push_down！）就能找到根。

xxxxxxxxxx
int FindRoot(int x){
    Access(x);splay(x);
    while(T[x].c[0]) push_down(x),x=T[x].c[0];
    splay(x);  //注意！
    return x;
}

注意标记了的那一行。对于LCT，一棵Splay的树根确实是可以随便选的，但是……我们在下面讲到Link,Cut操作的判断时再说。

4.Split 操作

Split(x,y)操作就是“提取”(x,y)这条链，使它成为一条实链，也就是一棵Splay，同时在我的代码里还设y为这棵Splay的根。

也很简单了：

xxxxxxxxxx
void Split(int x,int y){
    MakeRoot(x);Access(y);splay(y);
}

5,6.Link,Cut 操作

没有Link和Cut怎么能叫Link Cut Trees呢

先讲Link(x,y)，它的功能是在x,y之间建一条边。在我的代码里它有一个返回值：0表示x,y本来就连通，连边失败；否则连边成功并返回1。

如何判断x,y联通？把x设为所在原树的根，查看y的所在原树的根是否是x就可以了嘛。然后把x,y之间连一条虚边。

xxxxxxxxxx
bool Link(int x,int y){
    MakeRoot(x);
    if(FindRoot(y)==x) return 0;
    T[x].fa=y;
    return 1;
}

然后是Cut(x,y)，它的功能是把x,y之间的边割掉。同样也有返回值：0表示x,y之间原来没边，否则断边成功并返回1。

如何判断x,y有边？还是把x设为所在原树的边。首先，如果x,y不在一棵树内则肯定无边；

注意！这里调用了FindRoot(y)！你在FindRoot后，x,y会形成一条实链且y没有向儿子连出任何虚边，如果你不在之前“注意”的地方加splay(x)，那么这条实链的Splay的根会是y而不是x，无法进行下面的判断！

然后再判断，如果x,y有边那么现在x,y这条实链（其实就是(x,y)这一条边）长度必定为2，这是一种判断方法；

或者，如果x,y中间还有其他点，那么y的父亲就会不是x而是中间的某个点，或者y的左子树中有点。这也是一种判断方法。

然后使y不认这个父亲，x不认y这个儿子。

代码如下。这里使用第二种判断方法。看个人喜好了吧。

xxxxxxxxxx
bool Cut(int x,int y){
    MakeRoot(x);
    if(FindRoot(y)!=x||T[y].fa!=x||T[y].c[0]) return 0;
    T[y].fa=0;T[x].c[1]=0;
    push_up(x);
    return 1;
}

using namespace std;


struct Splay{
    int fa,c[2],val,sum,s;
    bool lzy;
}T[100005];
bool isroot(int x){
    return T[T[x].fa].c[0]!=x&&T[T[x].fa].c[1]!=x;
}
void push_up(int x){
    T[x].s=T[T[x].c[0]].s+T[T[x].c[1]].s+1;
    T[x].sum=T[T[x].c[0]].sum^T[T[x].c[1]].sum^T[x].val;
}
void reverse(int x){
    T[x].lzy^=1;
    swap(T[x].c[0],T[x].c[1]);
}
void push_down(int x){
    if(!T[x].lzy) return;
    T[x].lzy=0;
    if(T[x].c[0]) reverse(T[x].c[0]);
    if(T[x].c[1]) reverse(T[x].c[1]);
}
void rotate(int x){
    int y=T[x].fa,z=T[y].fa;
    bool k=(T[y].c[1]==x);
    if(!isroot(y)) T[z].c[T[z].c[1]==y]=x;
    T[x].fa=z;
    if(T[x].c[k^1]) T[T[x].c[k^1]].fa=y;
    T[y].c[k]=T[x].c[k^1];
    T[x].c[k^1]=y;
    T[y].fa=x;
    push_up(y);
}

7.简单应用：动态加边最小生成树

直接让当前的最小生成树被LCT所维护。假设现在要加一条边(u,v)，那么就会产生一个环：原来u→v的路径和(u,v)。然后我们在这个环上挑一条最大的边（比较原来u→v上最大的边和(u,v)）删掉，这样肯定最优。

然而，LCT维护的是点权不是边权，如何查询路径最大边权？对每条边额外设一个点，原图中的点点权为0而这种虚点的点权即为所代表的边权。用边i连接(u,v)的时候，实际连接的是(u,N+i),(v,N+i)。

P4172 [WC2006]水管局长（几乎）是板子，代码如下：

xxxxxxxxxx
#include<bits/stdc++.h>
using namespace std;


int N,M,Q;
struct Edge{
    int u,v,c;
    bool operator <(const Edge b)const{return u<b.u||(u==b.u&&v<b.v);}
}G[100005];
bool flg[100005];
int u[100005],v[100005],loc[100005],ans[100005];bool typ[100005];


struct Splay{
    int fa,c[2],val,sum,s;
    bool lzy;
}T[110005];
bool isroot(int x){
    return T[T[x].fa].c[0]!=x&&T[T[x].fa].c[1]!=x;
}
void push_up(int x){
    T[x].s=T[T[x].c[0]].s+T[T[x].c[1]].s+1;
    T[x].sum=0;
    if(G[T[T[x].c[0]].sum].c>G[T[x].sum].c) T[x].sum=T[T[x].c[0]].sum;
    if(G[T[T[x].c[1]].sum].c>G[T[x].sum].c) T[x].sum=T[T[x].c[1]].sum;
    if(G[T[x].val].c>G[T[x].sum].c) T[x].sum=T[x].val;
}
void reverse(int x){
    T[x].lzy^=1;
    swap(T[x].c[0],T[x].c[1]);
}
void push_down(int x){
    if(!T[x].lzy) return;
    T[x].lzy=0;
    if(T[x].c[0]) reverse(T[x].c[0]);
    if(T[x].c[1]) reverse(T[x].c[1]);
}
void rotate(int x){
    int y=T[x].fa,z=T[y].fa;
    bool k=(T[y].c[1]==x);
    if(!isroot(y)) T[z].c[T[z].c[1]==y]=x;
    T[x].fa=z;
    if(T[x].c[k^1]) T[T[x].c[k^1]].fa=y;
    T[y].c[k]=T[x].c[k^1];
    T[x].c[k^1]=y;
    T[y].fa=x;
    push_up(y);
}
int stk[100005],len;
void splay(int x){
    int y=x;len=0;
    while(!isroot(y)) stk[++len]=y,y=T[y].fa;
    stk[++len]=y;
    while(len) push_down(stk[len--]);
    while(!isroot(x)){
        int fx=T[x].fa,ffx=T[fx].fa;
        if(!isroot(fx))
            ((T[ffx].c[0]==fx)^(T[fx].c[0]==x))?rotate(x):rotate(fx);
        rotate(x);
    }
    push_up(x);
}


void Access(int x){
    for(int y=0;x;y=x,x=T[x].fa)
        splay(x),T[x].c[1]=y,push_up(x);
}
void MakeRoot(int x){
    Access(x);splay(x);reverse(x);
}
int FindRoot(int x){
    Access(x);splay(x);
    while(T[x].c[0]) push_down(x),x=T[x].c[0];
    splay(x);
    return x;
}
void Split(int x,int y){
    MakeRoot(x);Access(y);splay(y);
}
bool Link(int x,int y){
    MakeRoot(x);
    if(FindRoot(y)==x) return 0;
    T[x].fa=y;
    return 1;
}
bool Cut(int x,int y){
    MakeRoot(x);
    if(FindRoot(y)!=x||T[y].fa!=x||T[y].c[0]) return 0;
    T[y].fa=0;T[x].c[1]=0;
    push_up(x);
    return 1;
}


void add_E(int id){
    MakeRoot(G[id].u);
    if(FindRoot(G[id].v)!=G[id].u){Link(G[id].u,N+id);Link(G[id].v,N+id);return;}
    Split(G[id].u,G[id].v);
    int tgt=T[G[id].v].sum;
    if(G[tgt].c>G[id].c) Cut(G[tgt].u,N+tgt),Cut(G[tgt].v,N+tgt),Link(G[id].u,N+id),Link(G[id].v,N+id);
}


int main(){
    scanf("%d%d%d",&N,&M,&Q);
    for(int i=1;i<=M;i++){
        scanf("%d%d%d",&G[i].u,&G[i].v,&G[i].c);
        if(G[i].u>G[i].v) swap(G[i].u,G[i].v);
    }
    for(int i=1;i<=M;i++) T[N+i].val=T[N+i].sum=i;
    sort(G+1,G+M+1);
    for(int i=1;i<=Q;i++){
        int opt;scanf("%d%d%d",&opt,&u[i],&v[i]);
        typ[i]=opt-1;
        if(u[i]>v[i]) swap(u[i],v[i]);
        if(typ[i]){
            int l=1,r=M;
            while(l<r){
                int mid=(l+r)>>1;
                if(G[mid].u<u[i]||(G[mid].u==u[i]&&G[mid].v<v[i])) l=mid+1;
                else r=mid;
            }
            flg[loc[i]=r]=1;
        }
    }
    for(int i=1;i<=M;i++)if(!flg[i]) add_E(i);
    for(int i=Q;i>=1;i--)
        if(typ[i]) add_E(loc[i]);
        else{
            Split(u[i],v[i]);
            ans[i]=G[T[v[i]].sum].c;
        }
    for(int i=1;i<=Q;i++)if(!typ[i])
        printf("%d\n",ans[i]);
    
    return 0;
}
