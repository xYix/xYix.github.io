P4135 作诗

类型：需要设置ans[i][j]数组，方便查询两个整块之间的答案

相对于前一题要难很多。

首先我们定义cnt[i][x]是从第i个块开头到整个序列末尾，x出现了几次。显然第i块开头到第j块末尾x的出现次数就是cnt[i][x]−cnt[j+1][x]。

然后定义ans[i][j]是从第i块开头到第j块的答案。注意到每多统计一个块，甚至一个数都会导致答案变化，所以必须对任意两个块都求一次这个东西。

乍看之下这个ans数组必须N2求，然而并非如此，我们可以NN扫一遍求出cnt和ans，这样进行：

枚举i（块的编号）
设置tmp=0
枚举j（从i块的开头到序列末尾）
    如果cnt[i][A[j]]是偶数且不是0
    （也就是说，原来A[j]的数量是正偶数，对tmp有贡献，但是现在在j的位
    置又发现一个A[j]，cnt[i][A[j]]变为奇数，不再有贡献了）
        则tmp--
    如果cnt[i][A[j]]是奇数
    （也就是说，原来没有贡献但是把这个新发现的A[j]算上后就是偶数个
    了，具有贡献）
        则tmp++
    cnt[i][A[j]]++
    如果j是所在块的末尾
        ans[i][j所在块]=tmp

也就是说，发现一个A[j]就需要对它的贡献进行修正。询问时我们也会用到这种方法：

    （有贡献没算）

#include<bits/stdc++.h>
#define ll long long
using namespace std;


int N,M,LEN;
int C[50005],inb[50005];
int beg[230],cnt[230][50005];
ll ans[230][230];


int bok[50005];
int stk[50005],len;


ll gcd(ll a,ll b){
    return b?gcd(b,a%b):a;
}


int main(){
    scanf("%d%d",&N,&M);LEN=sqrt(N);
    for(int i=1;i<=N;i++){
        scanf("%d",&C[i]);
        inb[i]=(i-1)/LEN+1;
        if(inb[i]!=inb[i-1]) beg[inb[i]]=i;
    }
    beg[inb[N]+1]=N+1;inb[N+1]=inb[N]+1;
    for(int i=1;i<=inb[N];i++){
        ll tmp=0;
        for(int j=beg[i];j<=N;j++){
            tmp-=1LL*cnt[i][C[j]]*cnt[i][C[j]];
            cnt[i][C[j]]++;
            tmp+=1LL*cnt[i][C[j]]*cnt[i][C[j]];
            if(j+1==beg[inb[j]+1]) ans[i][inb[j]]=tmp;
        }
    }
    
    while(M--){
        int l,r;ll ANS=0;scanf("%d%d",&l,&r);
        if(l==r){printf("0/1\n");continue;}
        if(inb[l]==inb[r]){
            for(int i=l;i<=r;i++)
                bok[C[i]]++,stk[++len]=C[i];
            while(len){
                int x=stk[len--];
                if(!bok[x]) continue;
                ANS+=1LL*bok[x]*bok[x];
                bok[x]=0;
            }
        }
        else{
            ANS=ans[inb[l]+1][inb[r]-1];
            for(int i=l;i<beg[inb[l]+1];i++)
                bok[C[i]]++,stk[++len]=C[i];
            for(int i=beg[inb[r]];i<=r;i++)
                bok[C[i]]++,stk[++len]=C[i];
            while(len){
                int x=stk[len--];
                if(!bok[x]) continue;
                int T=cnt[inb[l]+1][x]-cnt[inb[r]][x];
                ANS-=1LL*T*T;ANS+=1LL*(T+bok[x])*(T+bok[x]);
                bok[x]=0;
            }
        }
        ANS-=(r-l+1);
        ll fm=1LL*(r-l+1)*(r-l),GCD=gcd(ANS,fm);
        printf("%lld/%lld\n",ANS/GCD,fm/GCD);
    }
    
    return 0;
}

P3203 [HNOI2010]弹飞绵羊

类型：较为独特

众所周知此题比较知名的原因是lct正解被分块吊起来打……

我们还是先分块，块长N。

我们维护两个信息：一个点要弹几次才能弹出所在块，而弹出来之后去了哪里。（听起来就非常暴力……）

O(N)扫一遍预处理，询问时一个块一个块跳，O(N)；修改时暴力直接把所在块内信息全改掉，O(N)。

时间复杂度 非 常 优 秀 ，考虑到常数、敲和调的难度的话真的是比lct高到不知道哪里去了

其实还是挺sb的……代码如下

#include<bits/stdc++.h>
using namespace std;


int N,len,M;
int K[200005];
int inb[200005],beg[450];
int cnt[200005],tgt[200005];


int main() {
    scanf("%d",&N);len=sqrt(N);
    for(int i=1;i<=N;i++){
        scanf("%d",&K[i]);
        inb[i]=(i-1)/len+1;
        if(inb[i]!=inb[i-1]) beg[inb[i]]=i;
    }
    for(int i=N;i>=1;i--){
        int pos=i+K[i];
        if(pos>N) cnt[i]=1,tgt[i]=N+1;
        else if(pos>=beg[inb[i]+1]) cnt[i]=1,tgt[i]=pos;
        else cnt[i]=cnt[pos]+1,tgt[i]=tgt[pos];

    while(M--){
        int opt;scanf("%d",&opt);
        if(opt==1){

                ans+=cnt[x];
                x=tgt[x];
            }
            printf("%d\n",ans);
        }
        else{
            int x,k;scanf("%d%d",&x,&k);x++;
            K[x]=k;
            for(int i=x;i>=beg[inb[x]];i--){
                int pos=i+K[i];
                if(pos>N) cnt[i]=1,tgt[i]=N+1;
                else if(pos>=beg[inb[i]+1]) cnt[i]=1,tgt[i]=pos;
                else cnt[i]=cnt[pos]+1,tgt[i]=tgt[pos];
            }
        }
    }
    
    return 0;
}

P2801 教主的魔法

类型：需要进行块内排序，修改时暴力重构

区间查询大于等于w的元素的个数可以这么搞：对于一个排好序的区间，我们可以直接二分出第一个≥w的元素然后算。又因为两个区间的答案可以直接加起来得出整个区间的答案，所以我们把序列分块，复制一个原序列的块内排好序的复制，然后查询的时候对每一个块内二分，边角料直接统计。

修改也就很好做了，完整的块打lazytag，不完整的块直接修改后重新排序。

可以发现，分块方法从单独的操作/询问来看都非常sb，但是其他优秀方法都不能兼容这几个操作/询问，于是分块就变得优秀了。

代码如下：

xxxxxxxxxx
#include <bits/stdc++.h>
using namespace std;


int N,M,len,H[1000005],inb[1000005];
int beg[1005],lzy[1005],cnt;
int H1[1000005];


int Find_ans(int i,int w){
    w-=lzy[i];
    int l=beg[i],r=beg[i+1],mid;
    while(l<r){
        mid=(l+r)>>1;
        if(H1[mid]>=w) r=mid;
        else l=mid+1;
    }
    return beg[i+1]-r;
}


int main(){
    scanf("%d%d",&N,&M);len=sqrt(N);
    for(int i=1;i<=N;i++){
        scanf("%d",&H[i]);H1[i]=H[i];
        inb[i]=(i-1)/len+1;
        if(inb[i]!=inb[i-1]) beg[inb[i]]=i,cnt++;
    }
    inb[N+1]=cnt+1;beg[cnt+1]=N+1;
    for(int i=1;i<=cnt;i++)
        sort(H1+beg[i],H1+beg[i+1]);
    
    while(M--){
        char opt=getchar();while(opt!='M'&&opt!='A') opt=getchar();
        if(opt=='M'){
            int l,r,w;scanf("%d%d%d",&l,&r,&w);
            if(inb[l]==inb[r]){
                for(int i=l;i<=r;i++) H[i]+=w;
                for(int i=beg[inb[l]];i<beg[inb[l]+1];i++)
                    H1[i]=H[i];
                sort(H1+beg[inb[l]],H1+beg[inb[l]+1]);
                continue;
            }
            for(int i=l;i<beg[inb[l]+1];i++) H[i]+=w;
            for(int i=beg[inb[l]];i<beg[inb[l]+1];i++)
                H1[i]=H[i];
            sort(H1+beg[inb[l]],H1+beg[inb[l]+1]);
            for(int i=beg[inb[r]];i<=r;i++) H[i]+=w;
            for(int i=beg[inb[r]];i<beg[inb[r]+1];i++)
                H1[i]=H[i];
            sort(H1+beg[inb[r]],H1+beg[inb[r]+1]);
            for(int i=inb[l]+1;i<=inb[r]-1;i++)
                lzy[i]+=w;
        }
        else{
            int l,r,w,ans=0;scanf("%d%d%d",&l,&r,&w);
            if(inb[l]==inb[r]){
                for(int i=l;i<=r;i++) ans+=(H[i]+lzy[inb[l]]>=w);
                printf("%d\n",ans);
                continue;
            }
            for(int i=l;i<beg[inb[l]+1];i++) ans+=(H[i]+lzy[inb[l]]>=w);
            for(int i=beg[inb[r]];i<=r;i++) ans+=(H[i]+lzy[inb[r]]>=w);
            for(int i=inb[l]+1;i<=inb[r]-1;i++) ans+=Find_ans(i,w);
            printf("%d\n",ans);
        }
    }
    
    return 0;
}

2.简单的莫队

因为莫队一般也分块所以我们讲一下莫队。

莫队的流程如下：

  • 首先把询问全读进来

  • 给询问安排一个处理顺序

  • 从询问i到i+1时，我们从li,ri一点一点挪到li+1,ri+1。

“一点一点挪”很重要。也就是如果l,r其中一个++或--，我们能方便地得出新的答案。

显然如果处理顺序不好，li,ri和li+1,ri+1差了很多的话莫队会变成常数很大的N2暴力。

一般来讲，我们安排顺序的时候，先分成一些块，然后把l所在块编号作为第一关键字，r（不是r所在块编号，就是r自己）作为第二关键字。

P3709 大爷的字符串题

这其实是一道语文题，我们跳过阅读理解环节，直接看翻译：

题目翻译：求区间众数的出现次数。（答案乘-1）

看上去正常方法和直接分块都不太可做，我们考虑一下莫队。考虑莫队也就是考虑如何处理挪一小步的问题。

我们定义cnt1[x]是x这个数在当前区间里出现了几次，cnt2[l]表示出现次数为l的数有几个。显然我们要先离散化。

如果我们把一个值为x的元素从区间里拿掉，那么cnt2[cnt1[x]]会--，这可以解释为x的出现次数-1，那么出现次数等于x原来的出现次数的元素个数肯定会少一个。同样还有cnt2[cnt1[x]+1]++，cnt1[x]--。

我们考虑这对答案有什么影响？如果x原来不是出现次数最多的元素（之一），那么显然对答案没有影响；如果x是出现次数最多的元素但不是唯一一个，那么还是没有影响；只有当x是唯一的出现次数最多的元素的时候，答案才会--。

同理我们也处理新加入x的情况。那么此题就做完了。

最后一点：注意到cnt1[]不能直接开下，于是离散化。

xxxxxxxxxx
#include <bits/stdc++.h>
using namespace std;


int N,M,LEN,N0;
int A[200005],A1[200005],inb[200005];
struct Queries{
    int l,r,id;
    bool operator <(const Queries b)const{
        return inb[l]<inb[b.l]||(inb[l]==inb[b.l]&&((r<b.r)^(inb[l]&1)));
    }
}Q[200005];


int cnt1[200005],cnt2[200005],ans;
int ANS[200005];
void inc(int x){
    cnt2[cnt1[x]]--;
    cnt1[x]++;
    cnt2[cnt1[x]]++;
    if(cnt1[x]>ans) ans=cnt1[x];
}
void dec(int x){
    cnt2[cnt1[x]]--;
    cnt1[x]--;
    cnt2[cnt1[x]]++;
    if(!cnt2[ans]) ans--;
}


int main(){
    scanf("%d%d",&N,&M);LEN=sqrt(N);
    for(int i=1;i<=N;i++) scanf("%d",&A[i]),A1[i]=A[i],inb[i]=(i-1)/LEN+1;
    sort(A1+1,A1+N+1);N0=unique(A1+1,A1+N+1)-A1-1;
    for(int i=1;i<=N;i++) A[i]=lower_bound(A1+1,A1+N0+1,A[i])-A1;
    
    for(int i=1;i<=M;i++) scanf("%d%d",&Q[i].l,&Q[i].r),Q[i].id=i;
    sort(Q+1,Q+M+1);
    
    int L=1,R=0;
    for(int i=1;i<=M;i++){
            //调整L,R到合适的位置
        while(R>Q[i].r) dec(A[R--]);
        while(R<Q[i].r) inc(A[++R]);
        while(L>Q[i].l) inc(A[--L]);
        while(L<Q[i].l) dec(A[L++]);
        ANS[Q[i].id]=ans;
    }
    for(int i=1;i<=M;i++)
        printf("-%d\n",ANS[i]);
    
    return 0;
}

P3674 小清新人渣的本愿

如果说，分块和莫队是优雅的暴力，那么bitset就是纯粹的暴力，然而悲哀的是它因为常数小所以能过。

首先莫队一下。这样我们可以维护一个桶表示每个数出现几次，然后考虑如何查询。

就是直接用bitset把那个桶（有就是1，无就是0）搞成一个大二进制数，然后这个数右移x位和自己取或如果有1出现，那么显然说明存在a−b=x。加法同理。除法则暴力枚举因数对。

下面考虑带修莫队。

我们给询问增加一维“时间维”，指的是这个询问前有多少个修改。排序的时候，l所在块为第一关键字，r所在块为第二关键字，c（时间维）为第三关键字。

我们还是和不带修莫队一样的思想，只不过，在两个询问跳转的时候，我们不仅要让l,r一个一个位置地移动，也要让c一个一个操作地移动。比如排序后的第一个询问前有3个操作，第二个有6个操作，那么我们转移的时候就要进行操作4,5,6，而反过来的话，我们就要执行4,5,6的逆操作。这样时间维已经统一，接下来再把l,r调整好。

P1903 [国家集训队]数颜色 / 维护队列

一个板子。关于逆操作，我们记录一个操作前该元素的值Kbef[i]，那么正向操作就是Kbef[i]→Kaft[i]，反向操作就是Kaft[i]→Kbef[i]。

#include <bits/stdc++.h>

int N,M,LEN;

            //调整操作次数

        while(ver>Qs[i].c){
            if(L<=Pos[ver]&&Pos[ver]<=R)
                inc(Kbef[ver]),dec(Kaft[ver]);
            A[Pos[ver]]=Kbef[ver];
            ver--;
        }
            //调整L,R

        while(R>Qs[i].r) dec(A[R--]);

如果我们把莫队搞到树上呢？也就是把区间改为链，然后做查询。

你可能会想到树剖——一个很经典的把树变成序列的方法。然而它不能满足我们的需要，因为树剖会把一条链转成log⁡n级别数量的区间，不合适。

这时我们引入欧拉序（或者叫括号序列）：在DFS过程中，访问到一个节点x，便在当前欧拉序末尾加入一个它的编号，这时的序列长度记作s[x]；在将要离开当前节点的时候也这么做，序列长度记作t[x]。（第一次加入和第二次加入恰好把此节点的整个子树“套住”了，括号序列这个名字挺贴切的）

注意到，我们离开一个子树之后，它里面的所有节点都会进入欧拉序两次，我们可以借助这个来判断一个节点应不应该有贡献。具体来说，在截取的括号序列里出现两次的节点不该有贡献。例子见下。

比如这棵树，我们现在想要对(u,v)进行操作。我们就应该：

  • 如果u,v都不是lca(u,v)，则截取[t[u],s[v]]，并特判lca(u,v)。

比如(5,4)这条链，我们截取[4,14]，也就是{5,6,6,2,3,7,8,8,7,3,4}，注意到5,4,2都出现了1次，它们确实是有贡献的点。然后特判lca(5,4)=1，整条链就成功截取出来了。

  • 否则，不妨设u=lca(u,v)，那么截取[s[u],s[v]]。

可以同样进行验证。

具体证明不展开讲，算法竞赛要什么证明
