可持久化数据结构，就是可以让你的每一次查询和修改都在某一历史版本上进行的数据结构。

可持久化数据结构的烧饼想法就是每次创建新版本时直接把旧版本整个复制过来，这样一来，时间和空间都无法承受。实际使用时，我们可以通过新版本和旧版本共享一些相同节点的方式来节约时间和空间。

1.可持久化字典树

先讲可持久化字典树吧，因为它简单。

众所周知，正常的字典树要插入一个字符串是这样的：

如果我们希望它可持久化（下称插入xex前的版本为旧版本，之后的为新版本），那么我们可以看下哪些点可以共用：

  • 对于新建的两个节点，肯定不能共用。

  • 根->t->c->l（注意是加粗的三个点）完全是可以共用的，新的插入的字符串跟它半毛钱关系都没有。

  • 根->x->y->x（注意是加粗的两个点）也是可以共用的，理由和上面差不多。

  • 对于根的x儿子节点，它不能共用。如果共用了，那么我们就可以在访问旧版本的时候，通过根->x->e->x访问到新插进的这个字符串xex，这是不合法的。我们需要给它建一个相似的复制。

  • 对于根，它也不能共用了。因为如果共用，我们就可以在旧版本通过根访问到 根旧的x儿子的复制 ，从而访问到xex，不合法。

你发现了什么没有？我们把整个“xex”路径上的节点都复制了一遍。

P4735 最大异或和

首先，看到区间异或立刻建前缀异或和。为什么不是后缀异或和呢？因为序列的末尾在增长，会很不好维护。

那么设S[i]=a[1] xor a[2]⋅⋅⋅xor a[i]，那么询问就是求

max(S[p−1] xor S[N] xor x)

不考虑[l,r]的限制，把S[N] xor x看成一个数，那么这就是一个非常经典的01trie问题。如果你不会这个问题，我建议你去找其他博客看一下。下面的讲解默认你对它的解法有较深理解。

然而毕竟还是有[l,r]的限制，我们怎么搞呢？

首先，我们可以给trie的节点附上一个lst[u]，表示u最后一次被走过——也就是u的子树最后一次被插入S[ ]是什么时候。如果lst[u]<l，那么u的子树里肯定都是不符合条件，<l的p。反之，如果lst[u]≥l，那么u的子树里一定有≥l的p。如果 一直 按这种规则走下去，走到的一定是≥l的p。

然而，还有一个限制是p≤r。这个怎么搞呢？我们不可能再搞一个类似的lst[ ]数组了（否则要反着插入数，而一个trie不可能又正着插又反着插）（疑车无据），那该怎么办呢？

不论什么样的标记都不合适（一个重要原因是trie的字符串之间有共用边，然而共用边是必须的，否则01trie的查询根本做不了），我们似乎很难不走那些p>r的边。

考虑一下：我们有没有方式，干脆回到一个根本没有插入过p>r的S[p]的时间点呢？这正是我们之前讲的可持久化字典树！

于是问题迎刃而解，请看代码：

#include <bits/stdc++.h>
using namespace std;


int N,M;
int RT[600005];
int S[600005];


struct Trie{
    int ch[2],lst;
}zero;vector<Trie> T;int idx;


void Insert(int X,int pos,int pos0,int i,int I){
    T.push_back(zero);pos0=++idx;T[pos0]=T[pos];T[pos0].lst=I;
    //开新点 
    //我的写法有些诡异，会有i=-1的点出现 
    if(i<0) return;
    bool flg=((X&(1<<i))>0);//提取这一位 
    T[pos0].ch[flg]=idx+1;  //建边(这个复制体的儿子也应是复制体) 
    Insert(X,T[pos].ch[flg],T[pos0].ch[flg],i-1,I);
}
int Query(int X,int pos,int i,int l){
    if(i<0) return 0;
    bool flg=((X&(1<<i))>0)^1;
    if(!T[pos].ch[flg]) return Query(X,T[pos].ch[flg^1],i-1,l);
    if(T[T[pos].ch[flg]].lst<l) return Query(X,T[pos].ch[flg^1],i-1,l);
    //我们提到的lst 
    return (1<<i)+Query(X,T[pos].ch[flg],i-1,l);
}


int main(){
    zero.ch[0]=zero.ch[1]=zero.lst=0;T.push_back(zero);
    scanf("%d%d",&N,&M);
    for(int i=1;i<=N;i++){
        int ti;scanf("%d",&ti);
        S[i]=S[i-1]^ti;
        RT[i]=idx+1;Insert(S[i],RT[i-1],RT[i],24,i);
        //注意此处插的是S[i]，但lst标记给的却是i，
        //所以在后面查询的时候要让l--
        //你喜欢的话当然可以在这里就-1 
    }
    for(int i=1;i<=M;i++){
        char c=getchar();while(c!='A'&&c!='Q') c=getchar();
        if(c=='A'){
            int ti;scanf("%d",&ti);
            N++;
            S[N]=S[N-1]^ti;
            RT[N]=idx+1;Insert(S[N],RT[N-1],RT[N],24,N);
        }
        else{
            int l,r,ti;scanf("%d%d%d",&l,&r,&ti);ti^=S[N];
            int ans=Query(ti,RT[r-1],24,l-1);
            if(l==1) ans=max(ans,ti);
            //注意S[0]=0我们是没有插进trie里的，要特判，你要插的话当然可以 
            printf("%d\n",ans);
        }
    }
    
    return 0;
}

P5283 [十二省联考2019]异或粽子

给出N+1个数S[i]=A[1] xor A[2] xor ... xor A[i]，求无序对(i,j)中S[i] xor S[j]最大的K对。

有点像P2048 [NOI2010]超级钢琴，思路是：

  • 一开始把所有i，对应的最大的(i,j)(j<i)塞进一个大根堆。

  • 弹出K次，如果i是第x次弹出就把第x+1大的(i,j)塞回堆里。

查询x大的(i,j)其实还是很好做的，有j<i的限制说明需要可持久化Trie树，具体操作就是把原来在Trie上走的过程稍微改一下，类比一下在权值线段树上查找第K大的过程就很好理解。

#include <bits/stdc++.h>
#define ll long long
using namespace std;


int N,K;
ll S[500005];


int ch[15000005][2],cnt[15000005],idx=1;
void Insert(ll x){
    int u=1;cnt[u]++;
    for(int k=31;k>=0;k--){
        bool flg=(x&((ll)1<<k))>0;
        if(!ch[u][flg]) ch[u][flg]=++idx;
        u=ch[u][flg];cnt[u]++;
    }
}
ll Query(ll x,int rk){
    int u=1;ll ans=0;
    for(int k=31;k>=0;k--){
        bool flg=(x&((ll)1<<k))>0;
        if(cnt[ch[u][flg^1]]<rk) rk-=cnt[ch[u][flg^1]],u=ch[u][flg];
        else ans+=((ll)1<<k),u=ch[u][flg^1];
    }
    return ans;
}


int cntk[500005];
struct node{
    int pos;ll val;
    bool operator <(const node b)const{return val<b.val;}
}H[500005];int len;
void putin(node x){H[++len]=x;push_heap(H+1,H+len+1);}
node putout(){pop_heap(H+1,H+len+1);return H[len--];}


ll ans;


int main(){
    scanf("%d%d",&N,&K);K*=2;
    Insert(0);
    for(int i=1;i<=N;i++){
        ll ai;scanf("%lld",&ai);
        S[i]=S[i-1]^ai;
        Insert(S[i]);
    }
    for(int i=0;i<=N;i++) cntk[i]=1,putin((node){i,Query(S[i],1)});
    for(int i=1;i<=K;i++){
        node u=putout();
        ans+=u.val;cntk[u.pos]++;
        if(cnt[u.pos]<=N+1) putin((node){u.pos,Query(S[u.pos],cntk[u.pos])});
    }
    printf("%lld",ans/2);
    
    return 0;
}

2.可持久化线段树

P4137 Rmq Problem / mex

我们可以开一个权值线段树（换句话说：我们用一棵线段树维护一个桶，放入值为A[i]的元素就是在位置A[i]处修改），参考我们最开始做的那道可持久化trie，我们给每个位置一个标记，表示最近一次修改是什么时候。

比如，我要询问区间[l,r]，发现在版本号为r的权值线段树中，有一个位置的标记≤l−1，那么久意味着这个位置表示的元素值在[l,r]中并没有出现。

于是权值线段树实际维护的是标记的min，询问的时候愉快二分。

#include <bits/stdc++.h>
using namespace std;


const int maxN=200005;


int N,M;
int A[maxN];


int Rot[maxN];
int Min[maxN*45],lc[maxN*45],rc[maxN*45],idx;
void push_up(int x){Min[x]=min(Min[lc[x]],Min[rc[x]]);}
int Build(int l,int r){
    int x=++idx;
    if(l!=r){
        int mid=(l+r)>>1;
        lc[x]=Build(l,mid);
        rc[x]=Build(mid+1,r); 
    }
    return x;
}
int Update(int x0,int l,int r,int pos,int K){
    int x=++idx;Min[x]=Min[x0];lc[x]=lc[x0];rc[x]=rc[x0];
    if(l==r){Min[x]=K;return x;}
    int mid=(l+r)>>1;
    if(pos<=mid) lc[x]=Update(lc[x0],l,mid,pos,K);
    else rc[x]=Update(rc[x0],mid+1,r,pos,K);
    push_up(x);
    return x;
}
int Query(int x,int l,int r,int K){
    if(!x) return l;
    if(l==r) return l;
    int mid=(l+r)>>1;
    if(Min[lc[x]]>=K) return Query(rc[x],mid+1,r,K);
    else return Query(lc[x],l,mid,K);
}


int main(){
    scanf("%d%d",&N,&M);
    Rot[0]=Build(1,N+1);
    for(int i=1;i<=N;i++){
        scanf("%d",&A[i]);
        if(A[i]<=N) Rot[i]=Update(Rot[i-1],1,N+1,A[i]+1,i);
        else Rot[i]=Rot[i-1];
    }
    while(M--){
        int l,r;scanf("%d%d",&l,&r);

    }

}

P3834 【模板】可持久化线段树

这一点都不模板……

有一个sb的想法（sb想法很多，但是这个比较容易优化），先把所有元素的值离散化，对于每个询问，我们把[l,r]的元素（当然这是被离散化过之后）全部放进一个权值线段树里，然后二分：

  • 如果左子树被插入的元素数量大于我们要查询的排名k，也就是说（离散化后的值）≤mid的元素都比k多了，那么说明该元素的值应在左子树表示的区间内，我们直接进入左子树。

  • 反之，该元素应在右子树内。考虑到右子树元素都比左子树大（废话，[mid+1,r]中的元素肯定比[l,mid]中的元素大），那么在右子树中排名为k−左子树元素个数的元素就是在整个区间内排名为k的元素。所以我们把k−=左子树元素个数并进入右子树。

虽然这个想法非常sb，但是我们可以优化一下：我们可以搞一个“前缀和”S[i]（都是一棵棵线段树），S[i]和S[i−1]的区别就是S[i]在A[i]（当然是离散化后）位置+1。

那么[l,r]的所有元素“之和”（所有对应的操作叠加起来）就应该是“S[r]−S[l−1]”这棵线段树。当然事实上我们并不真的需要这棵线段树：

  • 在“访问这棵线段树时”，我们同时访问S[r]和S[l−1]，这棵不存在的线段树的区间和就应该是S[r],S[l−1]中对应节点的区间和之差。

  • 跳到左节点时，对应的两个在S[r]和S[l−1]的指针分别跳到左节点。右节点亦然。

这看似是一个很妙的想法，一次查询log⁡n的级别（访问两棵线段树）。然而，它的空间开不下……n棵线段树显然是不可能存下的

吗？？？还记不记得S[i]和S[i−1]只差一个操作，这不就是可持久化线段树的两个版本吗！

我们观察一下上一题做法，其本质是两个“前缀和”相减得出一个权值线段树。这题中，每个节点的“前缀和”线段树是从根到这一节点的“前缀和”。那么，显然链(u,v)的权值线段树就应该是S[u]+S[v]−S[lca(u,v)]−S[fa[lca(u,v)]]。

#include <bits/stdc++.h>
using namespace std;


const int maxN=100005,maxLOG=20;


int N,M,n;
int A[maxN],A1[maxN];
int Rot[maxN];


int lnk[maxN];
int pre[maxN<<1],tgt[maxN<<1],cnt;
void add_E(int u,int v){
    pre[++cnt]=lnk[u],tgt[cnt]=v,lnk[u]=cnt;
}


int sum[maxN*maxLOG],lson[maxN*maxLOG],rson[maxN*maxLOG],idx;
int Build(int l,int r){
    int x=++idx;
    if(l!=r){
        int mid=(l+r)>>1;
        lson[x]=Build(l,mid);
        rson[x]=Build(mid+1,r);
    }
    return x;
}
int Update(int x0,int l,int r,int pos){
    int x=++idx;
    lson[x]=lson[x0];rson[x]=rson[x0];sum[x]=sum[x0]+1;
    if(l!=r){
        int mid=(l+r)>>1;
        if(pos<=mid) lson[x]=Update(lson[x0],l,mid,pos);
        else rson[x]=Update(rson[x0],mid+1,r,pos);
    }
    return x;
}


int dep[maxN],Fa[maxLOG][maxN];
void DFS(int x,int fa){
    dep[x]=dep[fa]+1;
    Rot[x]=Update(Rot[fa],1,n,lower_bound(A1+1,A1+n+1,A[x])-A1);
    
    Fa[0][x]=fa;
    for(int k=1;k<=19;k++)
        Fa[k][x]=Fa[k-1][Fa[k-1][x]];
    
    for(int e=lnk[x];e;e=pre[e])if(tgt[e]!=fa)
        DFS(tgt[e],x);
}


int LCA(int x,int y){
    if(dep[x]<dep[y]) swap(x,y);
    for(int k=19;k>=0;k--)
        if(dep[Fa[k][x]]>=dep[y]) x=Fa[k][x];
    if(x==y) return x;
    for(int k=19;k>=0;k--)
        if(Fa[k][x]!=Fa[k][y]) x=Fa[k][x],y=Fa[k][y];
    return Fa[0][x];
}


int Query(int x0,int x1,int x2,int x3,int l,int r,int K){
    if(l==r) return l;
    int X=sum[lson[x0]]+sum[lson[x1]]-sum[lson[x2]]-sum[lson[x3]];
    int mid=(l+r)>>1;
    if(X>=K) return Query(lson[x0],lson[x1],lson[x2],lson[x3],l,mid,K);
    else return Query(rson[x0],rson[x1],rson[x2],rson[x3],mid+1,r,K-X);
} 


int lst;


int main(){
    scanf("%d%d",&N,&M);
    for(int i=1;i<=N;i++) scanf("%d",&A[i]),A1[i]=A[i];
    sort(A1+1,A1+N+1);
    n=unique(A1+1,A1+N+1)-A1-1;
    for(int i=1;i<=N-1;i++){
        int u,v;scanf("%d%d",&u,&v);
        add_E(u,v);add_E(v,u);
    }
    
    Rot[0]=Build(1,n);
    DFS(1,0);
    
    while(M--){
        int a,b,c,d,K;scanf("%d%d%d",&a,&b,&K);a^=lst;
        c=LCA(a,b);d=Fa[0][c];
        printf("%d\n",lst=A1[Query(Rot[a],Rot[b],Rot[c],Rot[d],1,n,K)]);
    }
    
    return 0;
}

P2617 Dynamic Rankings

区间第K大带修改。

修改元素意味着我们需要重搞“前缀和”，编号为i的元素改了，S[i]⋯S[N]也全得改，没法承受。我们得用其他方法维护“前缀和”。

关键词：单点修改，在线查询“前缀和”。立刻想到线段树和树状数组！我们这里采用树状数组，因为好写常数小。

S[i]变为区间(i−lowbit(i),i]的区间和，于是我们可以用类似正常树状数组的方式找到所有需要被修改的S[j]（最多log⁡n个）并修改。修改方法：在原A[i]（离散化后）处-1，在新A[i]（离散化后）处+1。查询也变为在2log⁡n（一个log⁡n是把[1,l−1]拆成的log⁡n个S[j]，另一个log⁡n则是[1,r]）棵线段树上跳。

唯一需要注意的是，我们提到过“新A[i]（离散化后）”，所以我们需要离线并把所有可能出现的A[i]离散化。

最后再说一点，其实我们这时候在打的已经只是正常的动态开点线段树而已了，因为这时我们的前缀和已经用树状数组维护，累计时不是单纯地把两个版本减一减。（大概这个意思吧）

void Update(int &x,int l,int r,int pos,int K){
    if(!x) x=++idx;
    sum[x]+=K;
    if(l==r) return;
    int mid=(l+r)>>1;
    if(pos<=mid) Update(ls[x],l,mid,pos,K);
    else Update(rs[x],mid+1,r,pos,K);
}
void UPDATE(int x,int K){
    int pos=lower_bound(I+1,I+leni+1,A[x])-I;
    for(int x1=x;x1<=N;x1+=x1&-x1) Update(rt[x1],1,leni,pos,K);
}

int QUERY(int L,int R,int K){
    lent[0]=lent[1]=0;
    for(int R1=R;R1;R1-=R1&-R1) tmp[0][++lent[0]]=rt[R1];
    for(int L1=L-1;L1;L1-=L1&-L1) tmp[1][++lent[1]]=rt[L1];
    
    int l=1,r=leni,mid,SUM=0;
    while(l<r){
        SUM=0;
        mid=(l+r)>>1;
        for(int i=1;i<=lent[0];i++) SUM+=sum[ls[tmp[0][i]]]; 
        for(int i=1;i<=lent[1];i++) SUM-=sum[ls[tmp[1][i]]]; 
        if(K<=SUM){
            for(int i=1;i<=lent[0];i++) tmp[0][i]=ls[tmp[0][i]]; 
            for(int i=1;i<=lent[1];i++) tmp[1][i]=ls[tmp[1][i]]; 
            r=mid;
        }
        else{
            for(int i=1;i<=lent[0];i++) tmp[0][i]=rs[tmp[0][i]]; 
            for(int i=1;i<=lent[1];i++) tmp[1][i]=rs[tmp[1][i]]; 
            l=mid+1;K-=SUM;
        }
    }
    return l;
}

#include <bits/stdc++.h>
#define ll long long
using namespace std;


const int maxN=200005;
struct fhq_Treap{
    int s,l,r;ll val,sum;bool lzy;
}A[maxN*150];
int idx,rt[maxN];


int NewNode(ll v){idx++;A[idx].sum=A[idx].val=v;A[idx].l=A[idx].r=A[idx].lzy=0;A[idx].s=1;return idx;}
int CopyNode(int x){A[++idx]=A[x];return idx;}
void Push_up(int x){
    A[x].s=A[A[x].l].s+A[A[x].r].s+1;
    A[x].sum=A[A[x].l].sum+A[A[x].r].sum+A[x].val;
}
void Push_down(int x){
    if(!A[x].lzy) return;
    if(A[x].l) A[x].l=CopyNode(A[x].l);
    if(A[x].r) A[x].r=CopyNode(A[x].r);
    swap(A[x].l,A[x].r);
    if(A[x].l) A[A[x].l].lzy^=1;
    if(A[x].r) A[A[x].r].lzy^=1;
    A[x].lzy=0;
}
void Split(int x,int &L,int &R,int v){
    if(!x){L=R=0;return;}
    Push_down(x);
    if(A[A[x].l].s+1<=v) {L=CopyNode(x);Split(A[x].r,A[L].r,R,v-A[A[x].l].s-1);}
    else                 {R=CopyNode(x);Split(A[x].l,L,A[R].l,v);}
    if(L) Push_up(L);
    if(R) Push_up(R);
}
void Merge(int x,int y,int &M){
    if(!x||!y){M=x+y;return;}
    Push_down(x),Push_down(y);
    if(99999998%(A[x].s+A[y].s)<A[x].s)
        M=x,Merge(A[x].r,y,A[M].r);
    else
        M=y,Merge(x,A[y].l,A[M].l);
    Push_up(M);
}
