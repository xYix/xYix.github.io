0x10.分治FFT

考虑：P4721 【模板】分治 FFT。

优秀的nlogn做法

钦定g[0]=0。设F(x)是f[i]的生成函数，G(x)是g[i]的生成函数。于是显然有

F(x)=F(x)G(x)+1

F(x)=11−G(x)

于是变成了多项式求逆板子。

分治FFT

可以发现，后面的项必然基于前面的项的计算。

CDQ分治？！

还真是CDQ分治的思想。对于[l,r)，先分治左区间，那么左区间对于右区间的f[x](mid≤x<r)的贡献是

∑i=lmid−1f[i]⋅g[x−i]

可以用FFT快速计算。然后再分治右区间。

时间复杂度nlog2⁡n，代码如下：

#include<bits/stdc++.h>
using namespace std;


const int p=998244353,g=3,ig=332748118;
const int maxN=262144;


int N;
int inv[maxN+5];
int W[25][maxN+5],iW[25][maxN+5];
int x=1,len,X;
int G[maxN+5],F[maxN+5],tmpf[maxN+5],tmpg[maxN+5],tmp[maxN+5];


int qpow(int a,int k,int tt){
    int ans=1;
    while(k){
        if(k&1) ans=1LL*ans*a%tt;
        a=1LL*a*a%tt;
        k>>=1;
    }
    return ans;
}


void init(){
    x=1;while(x<(N<<1)) x<<=1,len++;
    inv[1]=1;
    for(int i=2;i<=x;i++)
        inv[i]=1LL*(p-p/i)*inv[p%i]%p;
    for(int i=1;i<=len+1;i++){
        W[i][0]=1;iW[i][0]=1;
        W[i][1]=qpow(g,(p-1)/(1<<i),p),
        iW[i][1]=qpow(ig,(p-1)/(1<<i),p);
        for(int j=2;j<(1<<(i-1));j++)
            W[i][j]=1LL*W[i][j-1]*W[i][1]%p,
            iW[i][j]=1LL*iW[i][j-1]*iW[i][1]%p;
    }
    X=x;
}


int R[maxN+5];
void NTT(int d[],bool flg,int n0){
    x=1;len=0;while(x<n0) x<<=1,len++;
    for(int i=0;i<x;i++){
        R[i]=(R[i>>1]>>1)|((i&1)<<(len-1));
        if(i<R[i]) swap(d[i],d[R[i]]);
    }
    for(int i=1,l=1;i<x;i<<=1,l++)
    for(int j=0;j<x;j+=(i<<1))
        for(int k=0;k<i;k++){
            int a0=d[j+k],a1=1LL*(flg?iW[l][k]:W[l][k])*d[j+i+k]%p;
            d[j+k]=(a0+a1)%p;d[j+i+k]=(a0-a1+p)%p;
        }
    if(flg)
        for(int i=0;i<x;i++)
            d[i]=1LL*d[i]*inv[x]%p;
}


void CDQ(int l,int r){
    if(l+1==r) return;
    int mid=(l+r)>>1;
    CDQ(l,mid);
    for(int i=0;i<mid-l;i++)
        tmpf[i]=F[i+l];
    for(int i=mid-l;i<((r-l)<<1);i++)
        tmpf[i]=0;
    for(int i=0;i<r-l;i++)
        tmpg[i]=G[i];
    for(int i=r-l;i<((r-l)<<1);i++)
        tmpg[i]=0;
    NTT(tmpf,0,(r-l)<<1);NTT(tmpg,0,(r-l)<<1);
    for(int i=0;i<((r-l)<<1);i++)
        tmp[i]=1LL*tmpf[i]*tmpg[i]%p;
    NTT(tmp,1,(r-l)<<1);
    for(int i=mid;i<r;i++) F[i]=(F[i]+tmp[i-l])%p;
    CDQ(mid,r);
}


int main(){
    scanf("%d",&N);init();
    for(int i=1;i<N;i++) scanf("%d",&G[i]);
    F[0]=1;
    CDQ(0,X>>1);
    for(int i=0;i<N;i++) printf("%d ",F[i]);
    
    return 0;
}

0x12.合并DFT

现在我们希望对两个多项式A,B进行DFT。但是我们不直接进行两次DFT，而是先设

{P=A+iBQ=A−iB

显然P,Q都可以O(N)得出。

为了方便书写，再设

{DFT(P)[k]=P(ωnk)DFT(Q)[k]=Q(ωnk)

那么显然有

DFT(P)[k]=A(ωnk)+iB(ωnk)=∑j=0n−1(a[j]+ib[j])ωnjk

还是为了方便书写，设X=2jkπn,conj(x)则表示x的共轭数。

DFT(P)[k]=∑j=0n−1(a[j]+ib[j])ωnjk=∑j=0n−1(a[j]+ib[j])(cos⁡X+isin⁡X)=conj(∑j=0n−1(a[j]−ib[j])(cos⁡X−isin⁡X))=conj(∑j=0n−1(a[j]−ib[j])(cos⁡(−X)+isin⁡(−X)))=conj(∑j=0n−1(a[j]−ib[j])ωn−jk)=conj(∑j=0n−1(a[j]−ib[j])ωnj(n−k))见证奇迹的时刻！=conj(DFT(Q)[n−k])

DFT出P就能O(N)得出Q，而A=P+Q2,B=P−Q2i，也就可以O(N)得出！两次DFT也就合并成了一次！

该方法被毛爷爷在2016年集训队论文里提到，该论文里还有一些其他的优化方法，非常值得去看一下。

0x13.多项式多点求值

构造多项式 P0=∏i=1n2(x−ai),P1=∏i=n2+1n(x−ai)，求出

F=Q0P0+R0=Q1P1+R1

显然对于 1≤i≤n2，F(ai)=R0(ai)，右半边同理。分治求解即可。总复杂度两只 log。

0x14.多项式快速插值

显然朴素的拉格朗日插值暴毙了，我们来看一下它怎么优化

∑iyi∏j≠i(x−xj)∏j≠i(xi−xj)

下面那个累乘对不同的 i 相差不大，所以目测可以一起处理

设

F(x)=∏j(x−xj)

那么下面那个累乘就是 F(x)x−xi 在（趋近） xi 的值，你发现这是一个 0/0 型的极限

于是我们请出【洛神】：求极限阶段时，你可以给一个函数的分子和分母求导，如果你还没求出极限而且它可以继续洛必达，则你可以再使用一次【洛神】

洛必达一下发现下面那个累乘就是 F′(xi)，要多点求值求出，下面记 yiF′(xi)=fi

现在要求的就是

∑ifi∏j≠i(x−xj)

设 ans[l,r] 是只考虑 [l,r] 之间的点求出的答案，显然有

ans[l,r]=ans[l,m]∏i=m+1r(x−xi)+ans[m+1,r]∏i=lm(x−xi)

显然那个累乘也是可以合并上来的

总复杂度两只 log，常数奇大无比。
