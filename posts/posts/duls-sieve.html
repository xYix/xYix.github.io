杜教筛可以搞出一个积性函数的前缀和：

S(N)=∑i=1Nf(i)

考虑f和另一个（随便选的）积性函数g的卷积：

∑i=1N(f∗g)(i)

当然也就是

∑i=1N∑d|if(id)∗g(d)

把d放在前面枚举，记j=id。

∑d=1N(g(d)∗∑j=1Ndf(j))

后面那个东西就是S(Nd)嘛。

也就是说，

∑i=1N(f∗g)(i)=∑d=1Ng(d)∗S(Nd)

又因为g是积性函数，所以g(1)=1。所以

∑i=1N(f∗g)(i)=∑d=2Ng(d)∗S(Nd)+S(N)

再化简单一点：

S(N)=∑i=1N(f∗g)(i)−∑d=2Ng(d)∗S(Nd)

这看上去很没有用，但是如果我们把g取得特殊一点就会很爽了。

比如P4213：

  • ∑i=1Nμ(i)

众所周知μ∗1=ϵ，所以如果把g取成1，那么第一个sigma就变成了1；第二个sigma则除法分块后可以递归地求解。

  • ∑i=1Nφ(i)

众所周知φ∗1=id，所以如果取1，那么第一个sigma就变成了N2+N2；第二个sigma一样求解。

递归可以写成记忆化搜索。怎么记录每个值访问过没有？注意到Ni最多N个（Nij=Nij，所以递归的任何一层都是Ni！），我们把大于N的值塞进一个根号大小的表（关键字就是Nx），小于的塞进一个根号大小的表。

这个复杂度看似sb但是它是O(N34)的。如果对较小的n（n≤N23）单独直接爆算出S(n)的话，还可以达到O(N23)。

#include<bits/stdc++.h>
#define ll long long
using namespace std;


int T,N,sqN;


const int len=2000000,LEN=50000;
bool flg[len+5];int pri[len+5],cnt;
int mu[len+5],phi[len+5],smu[len+5];ll sphi[len+5];
void init(){
    mu[1]=1;phi[1]=1;
    for(int i=2;i<=len;i++){
        if(!flg[i]) pri[++cnt]=i,mu[i]=-1,phi[i]=i-1;
        for(int j=1;j<=cnt&&i*pri[j]<=len;j++){
            flg[i*pri[j]]=1;
            
            if(i%pri[j]) mu[i*pri[j]]=-mu[i],phi[i*pri[j]]=phi[i]*(pri[j]-1);
            else {mu[i*pri[j]]=0;phi[i*pri[j]]=phi[i]*pri[j];break;}
        }
    }
    for(int i=1;i<=len;i++) smu[i]=smu[i-1]+mu[i],sphi[i]=sphi[i-1]+phi[i];
}


bool vis[LEN];int ans_mu[LEN];ll ans_phi[LEN];
int inline ret_mu(int x){
    if(x<=sqN) return smu[x];
    return ans_mu[N/x];
}
ll inline ret_phi(int x){
    if(x<=sqN) return sphi[x];
    return ans_phi[N/x];
}
void Solve(int x){
    if(x<=sqN) return;
    if(vis[N/x]) return;
    
    vis[N/x]=1;
    ans_mu[N/x]=1;ans_phi[N/x]=((ll)x*x+x)/2;
    for(int l=2,r;l<=x;l=r+1){
        r=(x/(x/l));
        Solve(x/l);
        ans_mu[N/x]-=(r-l+1)*ret_mu(x/l);
        ans_phi[N/x]-=(r-l+1)*ret_phi(x/l);
    }
}


int main(){
    scanf("%d",&T);
    init();
    while(T--){
        scanf("%d",&N);sqN=pow(N,2.0/3.0);
        memset(vis,0,sizeof(vis));
        Solve(N);
        printf("%lld %d\n",ret_phi(N),ret_mu(N));
    }
    
    return 0;
}
P3768 简单的数学题

常规操作，设gcd(i,j)=d并先枚举d。

∑d=1Nd3∑i=1Ndi∑j=1Ndj∗[gcd(i,j)==1]

反演一下

∑d=1Nd3∑i=1Ndi∑j=1Ndj∑g|i,g|jμ(g)

把g提到前面

∑d=1Nd3∑g=1Ndg2μ(g)∑i=1Ndgi∗∑j=1Ndgj

后面那个显然是

(Ndg)2(Ndg+1)24

于是最终得出：

∑d=1Nd3∑g=1Ndg2μ(g)(Ndg)2(Ndg+1)24

这个显然没法搞，还是进行常规操作，设T=dg，则可以写成

∑T=1N(NT)2(NT+1)24∑d|Td3μ(Td)∗(Td)2

写好看一点就是

∑T=1N(NT)2(NT+1)24T2∑d|Tdμ(Td)

你会发现，后面的那个

∑d|Tdμ(Td)

其实是个狄利克雷卷积（μ∗id），就是φ(T)。

然后记f(x)=x2φ(x)，不难看出这是一个积性函数。如果搞出这个东西的前缀和，那么我们前面那个东西就可以快乐地除法分块。由于数据范围太大，我们考虑杜教筛搞前缀和。

我们肯定希望

∑d|nf(d)g(nd)=∑d|nd2φ(d)g(nd)

简单一点。前面的那个d2非常令人不爽，于是我们设g(x)=x2，上面那个东西变成

n2∑d|nφ(d)

后面那个sigma是φ∗1=id，于是它就完美地变成了n3。

滚回去看杜教筛柿子。第一个sigma变成了立方和；第二个sigma也可以除法分块了。然后这题就做完了。

最后提一句，虽然要多次求f的前缀和，但是注意到我们只会调sum_f(r)和sum_f(l−1)，都是Ni，所以只需要调一次sum_f(N)就全求出来了。

#include<bits/stdc++.h>
#define ll long long
using namespace std;


int tt;ll N,sqN;
int inv4,inv6;
int qpow(int a,int k){
    int ans=1;
    while(k){
        if(k&1) ans=(1LL*ans*a)%tt;
        a=(1LL*a*a)%tt;
        k>>=1;
    }
    return ans;
}
int k2(ll l,ll r){
    l--;l%=tt;r%=tt;
    return (r%tt*(r+1)%tt*(r+r+1)%tt*inv6%tt-l%tt*(l+1)%tt*(l+l+1)%tt*inv6%tt+tt)%tt;
}
int k3(ll l,ll r){
    l--;l%=tt;r%=tt;
    return (r%tt*r%tt*(r+1)%tt*(r+1)%tt*inv4%tt-l%tt*l%tt*(l+1)%tt*(l+1)%tt*inv4%tt+tt)%tt;
}


const int len=8000000,LEN=200000;
bool flg[len+5];int pri[len+5],cnt;
int phi[len+5],sphi[len+5];
void init(){
    phi[1]=1;
    for(int i=2;i<=sqN;i++){
        if(!flg[i]) pri[++cnt]=i,phi[i]=i-1;
        for(int j=1;j<=cnt&&1LL*i*pri[j]<=sqN;j++){
            flg[i*pri[j]]=1;
            if(i%pri[j]) phi[i*pri[j]]=phi[i]*(pri[j]-1);
            else{phi[i*pri[j]]=phi[i]*pri[j];break;}
        }
    }
    for(int i=1;i<=sqN;i++) sphi[i]=((ll)sphi[i-1]+1LL*i*i%tt*phi[i]%tt)%tt;
}


bool vis[LEN+5];int ans[LEN+5];
int Get_sol(ll x){
    if(x<=sqN) return sphi[(int)x];
    return ans[(int)(N/x)];
}
void Solve(ll x){
    if(x<=sqN) return;
    int pos=(int)(N/x);
    if(vis[pos]) return;
    vis[pos]=1;ans[pos]=k3(1,x);
    for(ll l=2,r;l<=x;l=r+1){
        r=x/(x/l);
        Solve(x/l);
        ans[pos]=((ll)ans[pos]-1LL*k2(l,r)*Get_sol(x/l)%tt+tt)%tt;
    }
}


int Get_ANS(){
    int ANS=0;
    for(ll l=1,r;l<=N;l=r+1){
        r=N/(N/l);
        ANS=((ll)ANS+1LL*k3(1,N/l)*(((ll)Get_sol(r)-Get_sol(l-1)+tt)%tt)%tt)%tt;
    }
    return ANS;
}


int main(){ 
    scanf("%d%lld",&tt,&N);sqN=pow(N,2.0/3.0);inv4=qpow(4,tt-2);inv6=qpow(6,tt-2);
    init();
    Solve(N);
    printf("%lld\n",Get_ANS());


    return 0;
}
