#include <bits/stdc++.h>
using namespace std;


int N;
struct Point{
    long double x,y;
}A[1000005],O;


Point Solve_circle(Point a,Point b,Point c){
    long double A=b.x-a.x,B=b.y-a.y,C=c.x-b.x,D=c.y-b.y;
    long double E=b.x*b.x+b.y*b.y-a.x*a.x-a.y*a.y;
    long double F=c.x*c.x+c.y*c.y-b.x*b.x-b.y*b.y;
    return (Point){(F*B-E*D)/(C*B-A*D)/2.0,-(F*A-E*C)/(C*B-A*D)/2.0};
}
long double dis(Point a,Point b){
    return sqrt((a.x-b.x)*(a.x-b.x)+(a.y-b.y)*(a.y-b.y));
}


long double R;


int main(){
    scanf("%d",&N);
    for(int i=1;i<=N;i++) scanf("%Lf %Lf",&A[i].x,&A[i].y);
    random_shuffle(A+1,A+N+1);
    
    O=A[1];R=0;
    for(int i=1;i<=N;i++)if(dis(A[i],O)-R>1e-8){
        O=A[i];R=0;
        for(int j=1;j<i;j++)if(dis(A[j],O)-R>1e-8){
            O=(Point){(A[i].x+A[j].x)/2.0,(A[i].y+A[j].y)/2.0};
            R=dis(A[i],O);
            for(int k=1;k<j;k++)if(dis(A[k],O)-R>1e-8)
                O=Solve_circle(A[i],A[j],A[k]),
                R=dis(A[i],O);
        }
    }
    
    printf("%.10Lf\n%.10Lf %.10Lf",R,O.x,O.y);
    
    return 0;
}
