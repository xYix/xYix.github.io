变化#1：K度限制最小生成树

即要求某一顶点的度必须小于等于K。记该顶点为U。

遇到最小生成树类的问题，牢记一个生成树性质：

生成树上加一条边，一定有环。

太显然了，新加的边<u,v>和树上原来u,v的路径肯定是一个环吧。

然而这个性质是很有用的！！！切记

再回过头看这道题。

思考：U太特殊，先不考虑它，把其他顶点求一个最小生成森林。

先想清楚：各个树一定是孤立的，否则一定会合并为一棵树。接下来的把这几棵树合并一定通过特殊顶点这一中转站。

很显然吧？但还是千万记住。

设留下了M棵生成树。分类讨论：

  • M>K。肯定没法把这几棵树合并成一棵，无解。

  • M=K。每棵树和特殊顶点连一条最短的边。连不上去则无解。（即不连通）

  • M<K。麻烦就麻烦在这种情况：

首先每棵树和顶点连一条最短的边。先保证联通好，现在只剩一棵树了。

我们还有K−M条边可供支配。咋整呢？

从U往外连就完事了

嘛？连了之后，根据最开始叫你记的那个性质，这时“树”上一定有一个环。

如何把它变回树？从这个环里拿掉一条边就行了。拿那条边呢？贪心的想法，拿最长的那条嘛。

总结一下：

1.无视特殊顶点，Kruskal一趟求出最小生成森林。
2.如果M>K，无解。
3.每棵树到U连一条边。连不出来，即不连通，无解。
4.DP一趟求出每个顶点到U之间路径的最长边，记作A[v]。
5.枚举每个到U有不在当前生成树内的边（记作e）的顶点（记作v），val[e]−val[A[v]]记作v的价值。
6.拎出价值最小的v。如果它的价值≥0则不连，并直接结束算法。反正U的度≤K也没关系。否则断开A[v]，连上<U,v>。
7.如果当前U的度<K，则跳回4，否则结束算法。

完美！

#include<cstdio>
#include<iostream>
#include<map>
#include<cstring> 
#include<algorithm>
using namespace std;


struct Edge{
    int u,v,c;
    Edge(){
        u=v=c=0;
    }
    Edge(int u1,int v1,int c1) : u(u1),v(v1),c(c1){};
    bool operator <(const Edge b)const{return c<b.c;}
};


int N,M,K,ANS,Park;
map<string,int> Names;
string s1,s2;int u,v,c;


int bcj_fa[25];
int Find_fa(int x){
    return (bcj_fa[x]==x)?x:bcj_fa[x]=Find_fa(bcj_fa[x]);
}


Edge G1[1005];
int G2[25][25];bool in_tree[25][25];


void Kruskal(){
    sort(G1+1,G1+M+1);
    for(int i=1;i<=M;i++){
        int x1=G1[i].u,x2=G1[i].v;
        if(x1==Park || x2==Park) continue; 
        int f1=Find_fa(x1),f2=Find_fa(x2);
        if(f1==f2) continue;
        bcj_fa[f1]=f2;
        in_tree[x1][x2]=in_tree[x2][x1]=1;
        ANS+=G1[i].c;
    }
}
int Au[25],Av[25],Ac[25];
void DFS(int x,int fa){
    for(int i=1;i<=N;i++){
        if(i==x || i==fa) continue;
        if(G2[x][i]==-1) continue;
        if(!in_tree[x][i]) continue;
        if(Ac[i]!=-1) continue;
        
        Au[i]=Au[x];Av[i]=Av[x];Ac[i]=Ac[x];
        if(G2[x][i]>Ac[i]){
            Au[i]=x;Av[i]=i;Ac[i]=G2[x][i];
        }
        
        DFS(i,x);
    }
}


int M1;
int minE[25],keyP[25];


int main(){
    memset(G2,-1,sizeof(G2));
    cin>>M;
    for(int i=1;i<=M;i++){
        cin>>s1>>s2>>c;
        if(!Names.count(s1)) Names[s1]=++N;if(!Names.count(s2)) Names[s2]=++N;
        u=Names[s1],v=Names[s2];
        if(s1=="Park") Park=u;if(s2=="Park") Park=v;
        G1[i]=Edge(u,v,c);
        G2[v][u]=G2[u][v]=min((G2[u][v]==-1)?(1<<29):G2[u][v],c);
    }
    cin>>K;
    for(int i=1;i<=N;i++) bcj_fa[i]=i;
    Kruskal();
    memset(minE,-1,sizeof(minE));
    for(int i=1;i<=N;i++){
        if(i==Park) continue;
        if(G2[Park][i]==-1) continue;
        int clr=Find_fa(i);
        if(G2[Park][i]<minE[clr] || minE[clr]==-1) minE[clr]=G2[Park][i],keyP[clr]=i;
    } 
    for(int i=1;i<=N;i++){
        if(i==Park) continue;
        if(minE[i]==-1) continue;
        M1++;
        in_tree[Park][keyP[i]]=in_tree[keyP[i]][Park]=1;
        ANS+=minE[i]; 
    }
    for(int i=M1+1;i<=K;i++){
        memset(Au,-1,sizeof(Au));memset(Av,-1,sizeof(Av));memset(Ac,-1,sizeof(Ac));
        DFS(Park,Park);
        int pos=-1,Min_val=0;
        for(int j=2;j<=N;j++){
            if(G2[Park][j]==-1) continue;
            if(in_tree[Park][j]) continue;
            if(Min_val>G2[Park][j]-Ac[j]) Min_val=G2[Park][j]-Ac[j],pos=j;
        }
        if(Min_val>=0) break;
        in_tree[Park][pos]=in_tree[pos][Park]=1;
        in_tree[Au[pos]][Av[pos]]=in_tree[Av[pos]][Au[pos]]=1;
        ANS+=Min_val;
    }
    cout<<"Total miles driven: "<<ANS<<endl;
}

int Prim(){
    int ans=0;

变化#5：Sirni

显然就是一个最小生成树问题。

直接建图跑最小生成树显然不行，我们只好考虑改造一下原有的最小生成树算法。这里考虑Kruskal因为Prim考虑不出来。

实际上我们不需要把所有的边搞出来再排个序，而是直接从0开始枚举权值为x的边，这些边就是满足k∗P[u]+x=P[v]的(u,v)。

然而我们不需要固定一个u扫一遍所有的v，这会是SB级别的复杂度。我们不妨大胆猜想（大雾）P[i]是随机生成的，都很大而且差距不大，而且互不相同，那么我们可以考虑计数！固定一个u枚举所有的k，查看是否有对应的v，这样就找出了所有权值为x的边。然后按正常Kruskal做就可以了。

#include<bits/stdc++.h>
using namespace std;


int inline read(){
    int num=0,neg=1;char c=getchar();
    while(!isdigit(c)) {if(c=='-') neg=-1;c=getchar();}
    while(isdigit(c)) num=num*10+c-'0',c=getchar();
    return num*neg;
}


int N;
int P[100005],fa[10000005];
bool hsh[10000005];
long long ANS=0;
int L=1<<30,R;


int inline Find(int x){return fa[x]==x?x:fa[x]=Find(fa[x]);}


int main(){
    freopen("sirni.in","r",stdin);
    freopen("sirni.out","w",stdout);
    
    N=read();
    for(int i=1;i<=N;i++) {int t=read();hsh[t]=1,L=min(L,t),R=max(R,t);}
    N=0;
    for(int i=L;i<=R;i++)
        if(hsh[i])
            P[++N]=i,fa[i]=i;
    int k=N,md=0;
    while(1){
        if(k==1) break;
        for(int i=1;i<=N;i++){
            if(k==1) break;
            for(int j=P[i]+md;j<=R;j+=P[i]){
                if(!hsh[j]) continue;
                int fx=Find(P[i]),fy=Find(j);
                if(fx==fy) continue;
                fa[fx]=fy;ANS+=md;k--;
            }
        }
        md++;
    }
    
    printf("%lld",ANS);
    
    return 0;
}
