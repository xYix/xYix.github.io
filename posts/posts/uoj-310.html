void FWT(Z d[],bool flg){
    for(int n=1;n<N;n*=2)
    for(Z *j=d;j<d+N;j+=n*2)
    for(Z *k0=j,*k1=j+n,tmp;k0<j+n;k0++,k1++)
        tmp=*k1,*k1=*k0-*k1,*k0+=tmp;
